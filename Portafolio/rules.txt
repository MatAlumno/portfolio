CREATE DATABASE portfolio_db;
USE portfolio_db;

-- Usuario administrador (solo vos)
CREATE TABLE usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL
);

-- Info de la sección "Sobre mí"
CREATE TABLE datos_personales (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    descripcion TEXT,
    avatar_url VARCHAR(255),   -- imagen recortada / avatar
    fondo_url VARCHAR(255)     -- imagen background grande
);

-- Sección trayecto / educación / experiencia
CREATE TABLE trayecto (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(100),
    descripcion TEXT,
    fecha_inicio DATE,
    fecha_fin DATE
);

-- Proyectos mostrados en la página de proyectos
CREATE TABLE proyectos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    descripcion TEXT,
    link VARCHAR(255)          -- link al repo o preview
);

-- Contenido flexible para textos sueltos por página
CREATE TABLE contenido (
    id INT AUTO_INCREMENT PRIMARY KEY,
    seccion VARCHAR(50a,       -- index | sobre_mi | personal | trayecto | proyectos
    titulo VARCHAR(100),
    texto TEXT
);
